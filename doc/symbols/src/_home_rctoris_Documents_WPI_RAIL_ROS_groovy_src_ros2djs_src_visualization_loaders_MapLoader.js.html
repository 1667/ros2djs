  5 ROS2D.MapLoader = function(options) {
  6   options = options || {};
  7   this.ros = options.ros;
  8   this.viewer = options.viewer;
  9 };
 10 ROS2D.MapLoader.prototype.loadTopic = function(topic, continuous) {
 11   var that = this;
 12   var client = new ROS2D.MapClient({
 13     ros : this.ros,
 14     topic : topic,
 15     continuous : continuous
 16   });
 17   
 18   client.on('ready', function(image) {
 19     var bitmap = new createjs.Bitmap(image);
 20     bitmap.addEventListener("mousedown", function(){
 21       console.log('test');
 22     });
 23     that.viewer.addObject(bitmap);
 24   });
 25 };
