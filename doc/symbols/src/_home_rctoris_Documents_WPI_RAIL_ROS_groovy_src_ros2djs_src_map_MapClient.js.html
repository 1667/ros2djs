  5 ROS2D.MapClient = function(options) {
  6   var that = this;
  7   var options = options || {};
  8   this.ros = options.ros;
  9   this.topic = options.mapTopic || '/map';
 10   this.continuous = options.continuous;
 11 
 12   // map metadata
 13   this.pose = null;
 14   this.resolution = null;
 15   // internal drawing canvas
 16   this.image = document.createElement('canvas');
 17 
 18   // setup a listener for the map data
 19   var mapListener = new ROSLIB.Topic({
 20     ros : this.ros,
 21     name : this.topic,
 22     messageType : 'nav_msgs/OccupancyGrid',
 23     compression : 'png'
 24   });
 25   mapListener.subscribe(function(occupancyGrid) {
 26     var context = that.image.getContext('2d');
 27 
 28     // save the metadata we need
 29     that.pose = new ROSLIB.Pose({
 30       position : occupancyGrid.info.origin.position,
 31       orientation : occupancyGrid.info.origin.orientation
 32     });
 33     that.resolution = occupancyGrid.info.resolution;
 34 
 35     // set the size
 36     var width = occupancyGrid.info.width;
 37     var height = occupancyGrid.info.height;
 38     that.image.width = width;
 39     that.image.height = height;
 40 
 41     var imageData = context.createImageData(width, height);
 42     for ( var row = 0; row < height; row++) {
 43       for ( var col = 0; col < width; col++) {
 44         // determine the index into the map data
 45         var mapI = col + ((height - row - 1) * width);
 46         // determine the value
 47         var data = occupancyGrid.data[mapI];
 48         if (data === 100) {
 49           var val = 0;
 50         } else if (data === 0) {
 51           var val = 255;
 52         } else {
 53           var val = 127;
 54         }
 55 
 56         // determine the index into the image data array
 57         var i = (col + (row * width)) * 4;
 58         // r
 59         imageData.data[i] = val;
 60         // g
 61         imageData.data[++i] = val;
 62         // b
 63         imageData.data[++i] = val;
 64         // a
 65         imageData.data[++i] = 255;
 66       }
 67     }
 68     context.putImageData(imageData, 0, 0);
 69 
 70     // check if we only wanted one message
 71     if (!that.continuous) {
 72       mapListener.unsubscribe();
 73     }
 74 
 75     // notify the user an image is ready
 76     that.emit('ready', that.image);
 77   });
 78 };
 79 ROS2D.MapClient.prototype.__proto__ = EventEmitter2.prototype;
